Unit 7: Advance Android Concepts

Provided that a SQLite database named "College" with a table named Student with the following columns (Roll as Integer, Name as Text and Address as Text). Develop an android application to connect to the database and insert five student records and display the student information.

Full Code Example:

1. `DatabaseHelper.java`:

import android.content.ContentValues;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;

public class DatabaseHelper extends SQLiteOpenHelper {
    private static final String DATABASE_NAME = "College.db";
    private static final int DATABASE_VERSION = 1;
    private static final String TABLE_STUDENT = "Student";
    private static final String COLUMN_ROLL = "Roll";
    private static final String COLUMN_NAME = "Name";
    private static final String COLUMN_ADDRESS = "Address";

    public DatabaseHelper(Context context) {
        super(context, DATABASE_NAME, null, DATABASE_VERSION);
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        String CREATE_STUDENT_TABLE = "CREATE TABLE " + TABLE_STUDENT + "("
                + COLUMN_ROLL + " INTEGER PRIMARY KEY,"
                + COLUMN_NAME + " TEXT,"
                + COLUMN_ADDRESS + " TEXT" + ")";
        db.execSQL(CREATE_STUDENT_TABLE);
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
        db.execSQL("DROP TABLE IF EXISTS " + TABLE_STUDENT);
        onCreate(db);
    }

    public void addStudent(int roll, String name, String address) {
        SQLiteDatabase db = this.getWritableDatabase();
        ContentValues values = new ContentValues();
        values.put(COLUMN_ROLL, roll);
        values.put(COLUMN_NAME, name);
        values.put(COLUMN_ADDRESS, address);
        db.insert(TABLE_STUDENT, null, values);
        db.close();
    }

    public Cursor getAllStudents() {
        SQLiteDatabase db = this.getReadableDatabase();
        return db.rawQuery("SELECT * FROM " + TABLE_STUDENT, null);
    }
}

3. `MainActivity.java`:

public class MainActivity extends AppCompatActivity {
    private DatabaseHelper dbHelper;
    private TextView tvStudentData;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        dbHelper = new DatabaseHelper(this);
        tvStudentData = findViewById(R.id.tv_student_data);
        Button btnAddData = findViewById(R.id.btn_add_data);

        btnAddData.setOnClickListener(v -> {
            insertSampleData();
            displayStudentData();
        });
    }

    private void insertSampleData() {
        dbHelper.addStudent(1, "Alice", "123 Main St");
        dbHelper.addStudent(2, "Bob", "456 Oak Ave");
        dbHelper.addStudent(3, "Charlie", "789 Pine Ln");
        dbHelper.addStudent(4, "David", "101 Maple Dr");
        dbHelper.addStudent(5, "Eve", "212 Birch Rd");
    }

    private void displayStudentData() {
        Cursor cursor = dbHelper.getAllStudents();
        StringBuilder data = new StringBuilder();

        if (cursor.moveToFirst()) {
            do {
                int roll = cursor.getInt(cursor.getColumnIndexOrThrow("Roll"));
                String name = cursor.getString(cursor.getColumnIndexOrThrow("Name"));
                String address = cursor.getString(cursor.getColumnIndexOrThrow("Address"));
                data.append("Roll: ").append(roll)
                    .append(", Name: ").append(name)
                    .append(", Address: ").append(address).append("\n\n");
            } while (cursor.moveToNext());
        }
        cursor.close();
        tvStudentData.setText(data.toString());
    }
}

What are different data types supported by SQLite? How to decode JSON data in an android application? Explain with a suitable example.

SQLite Data Types (Type Affinity):

SQLite uses a dynamic type system. The data type of a value is associated with the value itself, not with its container (the column). However, SQLite does have a concept of "Type Affinity" on columns, which is the recommended type of data to be stored in that column. The five affinities are:

  • TEXT: Stores text strings.
  • NUMERIC: Stores all numerical values, including integers and floating-point numbers.
  • INTEGER: Similar to NUMERIC but with a preference for storing values as integers if possible.
  • REAL: Stores floating-point values.
  • BLOB (Binary Large Object): Stores any kind of data exactly as it was input (e.g., images, files).

Decoding JSON in Android:

Android provides the built-in `org.json` library to parse JSON data. You use `JSONObject` to represent a JSON object (a set of key-value pairs) and `JSONArray` for a JSON array (an ordered sequence of values).

Example:

// Sample JSON string
String jsonString = "{\n" +
        "  \"name\": \"John Doe\",\n" +
        "  \"age\": 30,\n" +
        "  \"isStudent\": false,\n" +
        "  \"courses\": [\n" +
        "    {\"title\": \"History\", \"credits\": 3},\n" +
        "    {\"title\": \"Math\", \"credits\": 4}\n" +
        "  ]\n" +
        "}";

try {
    // Create a JSONObject from the string
    JSONObject person = new JSONObject(jsonString);

    // Extract values by key
    String name = person.getString("name");
    int age = person.getInt("age");
    boolean isStudent = person.getBoolean("isStudent");

    System.out.println("Name: " + name);
    System.out.println("Age: " + age);
    System.out.println("Is Student: " + isStudent);

    // Get the JSONArray for courses
    JSONArray coursesArray = person.getJSONArray("courses");

    // Loop through the array
    for (int i = 0; i < coursesArray.length(); i++) {
        JSONObject course = coursesArray.getJSONObject(i);
        String title = course.getString("title");
        int credits = course.getInt("credits");
        System.out.println("Course: " + title + ", Credits: " + credits);
    }

} catch (JSONException e) {
    e.printStackTrace();
}
